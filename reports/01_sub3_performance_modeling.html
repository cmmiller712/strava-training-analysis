w["volume_score"] = w["miles"] / 60
w["long_run_score"] = w["long_run_miles"] / 20
w["specificity_score"] = w["mp_specificity"] / 0.2
w["ramp_penalty"] = 1 - w["ramp_rate"].clip(lower=0, upper=0.3)

w["sub3_confidence"] = (
    0.3 * w["volume_score"] +
    0.3 * w["long_run_score"] +
    0.3 * w["specificity_score"] +
    0.1 * w["ramp_penalty"]
)
